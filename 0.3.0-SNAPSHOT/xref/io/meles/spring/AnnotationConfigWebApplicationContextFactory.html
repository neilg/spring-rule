20  package io.meles.spring;
21  
22  import java.util.Arrays;
23  import java.util.Collections;
24  import java.util.HashMap;
25  import java.util.Map;
26  import java.util.Map.Entry;
27  
28  import org.springframework.beans.factory.support.DefaultListableBeanFactory;
29  import org.springframework.web.context.support.AnnotationConfigWebApplicationContext;
30  
31  public class AnnotationConfigWebApplicationContextFactory implements ApplicationContextFactory<AnnotationConfigWebApplicationContext> {
32      private final Class<?>[] configClasses;
33      private final Map<String, Object> beans;
34  
35      public AnnotationConfigWebApplicationContextFactory(final Class<?>[] configClasses, final Map<String, Object> beans) {
36          this.configClasses = copy(configClasses);
37          this.beans = Collections.unmodifiableMap(new HashMap<>(beans));
38      }
39  
40      @Override
41      public AnnotationConfigWebApplicationContext newInstance() {
42          final AnnotationConfigWebApplicationContext applicationContext = new AnnotationConfigWebApplicationContext() {
43              @Override
44              protected void loadBeanDefinitions(final DefaultListableBeanFactory beanFactory) {
45                  registerSingletons(beanFactory);
46                  super.loadBeanDefinitions(beanFactory);
47              }
48          };
49  
50          try {
51              registerConfig(applicationContext);
52          } catch (RuntimeException e) {
53              applicationContext.close();
54              throw e;
55          }
56  
57          return applicationContext;
58      }
59  
60      private DefaultListableBeanFactory registerSingletons(final DefaultListableBeanFactory factory) {
61          for (final Entry<String, Object> bean : beans.entrySet()) {
62              factory.registerSingleton(bean.getKey(), bean.getValue());
63          }
64          return factory;
65      }
66  
67      private void registerConfig(final AnnotationConfigWebApplicationContext applicationContext) {
68          if (configClasses.length > 0) {
69              applicationContext.register(copy(configClasses));
70          }
71          applicationContext.refresh();
72      }
73  
74      private static <T> T[] copy(T[] array) {
75          return Arrays.copyOf(array, array.length);
76      }
77  }
