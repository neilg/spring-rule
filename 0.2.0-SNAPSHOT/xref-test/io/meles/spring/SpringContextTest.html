42  public class SpringContextTest {
43  
44      @Rule
45      public ExpectedException expectedException = ExpectedException.none();
46  
47      private SpringContext springContext;
48      private SpringContext badSpringContext;
49  
50      @Before
51      public void createRules() {
52          springContext = SpringContext.builder()
53                  .withConfig(SimpleConfig.class)
54                  .addConfig(ExtraConfig.class)
55                  .build();
56          badSpringContext = SpringContext.builder()
57                  .withConfig(ThrowingConfig.class)
58                  .build();
59      }
60  
61      @Test
62      public void canRetrieveBeanFromContext() throws Throwable {
63          final Object[] holder = new Object[1];
64          springContext.apply(new Statement() {
65              @Override
66              public void evaluate() {
67                  holder[0] = springContext.getApplicationContext().getBean("stringBean");
68              }
69          }, Description.EMPTY).evaluate();
70          assertEquals("it's a string", holder[0]);
71      }
72  
73      @Test
74      public void canGetBeanFromAddedConfig() throws Throwable {
75          final Object[] holder = new Object[1];
76          springContext.apply(new Statement() {
77              @Override
78              public void evaluate() {
79                  holder[0] = springContext.getApplicationContext().getBean("extraString");
80              }
81          }, Description.EMPTY).evaluate();
82          assertEquals("the extra string", holder[0]);
83      }
84  
85      @Test
86      public void underlyingStatementIsEvaluated() throws Throwable {
87          final Statement statement = mock(Statement.class);
88          final Statement wrappedStatement = springContext.apply(statement, Description.EMPTY);
89          wrappedStatement.evaluate();
90          verify(statement, times(1)).evaluate();
91      }
92  
93      @Test
94      public void underlyingStatementIsNotEvaluatedIfContextFailsToStart() throws Throwable {
95          final Statement statement = mock(Statement.class);
96          final Statement wrappedStatement = badSpringContext.apply(statement, Description.EMPTY);
97          try {
98              wrappedStatement.evaluate();
99          } catch (Throwable ignored) {
100         } finally {
101             verify(statement, never()).evaluate();
102         }
103     }
104 
105     @Test
106     public void failureToStartIsPropagatedOnEvaluation() throws Throwable {
107         final Statement statement = mock(Statement.class);
108         final Statement wrappedStatement = badSpringContext.apply(statement, Description.EMPTY);
109         expectedException.expect(rootCause(hasMessage(equalTo("bad, bad, bad"))));
110         wrappedStatement.evaluate();
111     }
112 
113     @Test
114     public void cannotGetApplicationContextOutsideOfEvaluation() {
115         expectedException.expect(IllegalStateException.class);
116         expectedException.expectMessage("no spring application context, are you calling getApplicationContext() outside of a test execution");
117         springContext.getApplicationContext();
118     }
119 
120     @Test
121     public void contextIsClosedAfterEvaluation() throws Throwable {
122         final ApplicationContext[] holder = new ApplicationContext[1];
123         springContext.apply(new Statement() {
124             @Override
125             public void evaluate() {
126                 holder[0] = springContext.getApplicationContext();
127             }
128         }, Description.EMPTY).evaluate();
129         assertFalse(((ConfigurableApplicationContext) holder[0]).isActive());
130         assertFalse(((ConfigurableApplicationContext) holder[0]).isRunning());
131     }
132 
133     private Matcher<Throwable> rootCause(final Matcher<? super Throwable> throwableMatcher) {
134         return new TypeSafeDiagnosingMatcher<Throwable>() {
135             @Override
136             protected boolean matchesSafely(Throwable item, org.hamcrest.Description mismatchDescription) {
137                 Throwable cause = item;
138                 while (cause.getCause() != null) {
139                     cause = cause.getCause();
140                 }
141                 final boolean matches = throwableMatcher.matches(cause);
142                 if (!matches) {
143                     mismatchDescription.appendText("root cause ");
144                     throwableMatcher.describeMismatch(cause, mismatchDescription);
145                 }
146                 return matches;
147             }
148 
149             @Override
150             public void describeTo(org.hamcrest.Description description) {
151                 description.appendText("throwable with root cause ").appendDescriptionOf(throwableMatcher);
152             }
153         };
154     }
