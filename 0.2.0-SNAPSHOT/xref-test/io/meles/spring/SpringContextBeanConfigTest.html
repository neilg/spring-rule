20  package io.meles.spring;
21  
22  import static org.junit.Assert.assertEquals;
23  import static org.mockito.Mockito.mock;
24  
25  import org.junit.Before;
26  import org.junit.Rule;
27  import org.junit.Test;
28  import org.junit.rules.ExpectedException;
29  import org.junit.runner.Description;
30  import org.junit.runners.model.Statement;
31  
32  public class SpringContextBeanConfigTest {
33  
34      @Rule
35      public ExpectedException expectedException = ExpectedException.none();
36  
37      private SpringContext springContext;
38  
39      @Before
40      public void createRules() {
41          springContext = SpringContext.builder()
42                  .config(SimpleConfig.class)
43                  .config(ExtraConfig.class)
44                  .build();
45      }
46  
47      @Test
48      public void canBuildASpringContextWithNoConfig() throws Throwable {
49          final Statement statement = mock(Statement.class);
50          SpringContext.builder()
51                  .build()
52                  .apply(statement, Description.EMPTY)
53                  .evaluate();
54      }
55  
56      @Test
57      public void canRetrieveBeanFromContext() throws Throwable {
58          final Object[] holder = new Object[1];
59          springContext.apply(new Statement() {
60              @Override
61              public void evaluate() {
62                  holder[0] = springContext.getApplicationContext().getBean("stringBean");
63              }
64          }, Description.EMPTY).evaluate();
65          assertEquals("it's a string", holder[0]);
66      }
67  
68      @Test
69      public void canGetBeanFromAddedConfig() throws Throwable {
70          final Object[] holder = new Object[1];
71          springContext.apply(new Statement() {
72              @Override
73              public void evaluate() {
74                  holder[0] = springContext.getApplicationContext().getBean("extraString");
75              }
76          }, Description.EMPTY).evaluate();
77          assertEquals("the extra string", holder[0]);
78      }
79  
80  }
